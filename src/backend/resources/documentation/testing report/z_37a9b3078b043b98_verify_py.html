40def check_verification( 

41 app: App = Depends(App.get_instance), 

42 auth_token: str = Cookie(""), 

43): 

44 """ 

45 Check if the currently authenticated user has verified their email. 

46 

47 Parameters: 

48 - app (App): Dependency-injected application context providing access to services. 

49 - auth_token (str): Authentication token retrieved from the user's cookie. 

50 

51 Returns: 

52 - JsonResponseWithStatus: User verification status or appropriate error message. 

53 """ 

54 redis_client = app.get_redis_manager() 

55 db_session = app.get_db_session() 

56 try: 

57 # Retrieve authentication information from Redis 

58 auth_info = redis_client.get("auth_token", auth_token) 

59 

60 # Return 401 if the token is invalid or lacks user_id 

61 if auth_info is None or not auth_info.get("user_id"): 

62 return JsonResponseWithStatus( 

63 status_code=401, 

64 content=InvalidOrExpiredAuthToken(), 

65 ) 

66 

67 user_id = auth_info["user_id"] 

68 

69 # Retrieve user from database using user_id 

70 user = crud.get_user_by_id(db_session, user_id) 

71 

72 # Return 404 if user is not found 

73 if not user: 

74 return JsonResponseWithStatus( 

75 status_code=404, 

76 content=UserNotFoundError(), 

77 ) 

78 

79 # Return the verification status of the user 

80 return JsonResponseWithStatus( 

81 status_code=200, 

82 content=GetVerificationGetResponse(user_is_verified=user.verified), 

83 ) 

84 except Exception as e: 

85 # Log and return 500 on unexpected failure
